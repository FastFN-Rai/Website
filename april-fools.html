日 時間 分 秒 
JS Code

const now = new Date();
var date = new Date(now.getFullYear(), 4, 1, 0, 0, 0);
        
const fkdiffer = date.getTime() - now.getTime();
const fksec = Math.floor(fkdiffer / 1000) % 60;
        
if (Math.sign(fksec) == -1) {
    date = new Date(now.getFullYear() + 1, 4, 1, 0, 0, 0);
}
const differ = date.getTime() - now.getTime();

const sec = Math.floor(differ / 1000) % 60;
const min = Math.floor(differ / 1000 / 60) % 60;
const hour = Math.floor(differ / 1000 / 60 / 60) % 24;
const day = Math.floor(differ / 1000 / 60 / 60 / 24 );

console.log(`${day}日、${hour}時間、${min}分、${sec}秒`)
